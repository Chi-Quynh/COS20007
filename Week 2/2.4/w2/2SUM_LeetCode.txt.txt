public class Solution {
    public int[] TwoSum(int[] nums, int target) {
        var dict = new Dictionary<int,int>();
        int remain;

        
        for (int i=0;i<nums.Length;i++){
            remain = target - nums[i];
            if (dict.ContainsKey(remain)){
                return new int[]{i, dict[remain]};
            }
            else{
                dict[nums[i]] = i;
            }


        }
        return new int[]{};
    }
}
